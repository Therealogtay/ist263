// Listen for the DOM to be fully loaded document.addEventListener("DOMContentLoaded", function() { // Function to check if an element is in view function isInView(el) { var rect = el.getBoundingClientRect(); return ( rect.top >= 0 && rect.left >= 0 && rect.bottom <= (window.innerHeight || document.documentElement.clientHeight) && rect.right <= (window.innerWidth || document.documentElement.clientWidth) ); } // Function to animate elements on scroll function animateOnScroll() { document.querySelectorAll('.fadeInUp').forEach(function(el) { if (isInView(el)) { el.classList.add('visible'); } }); } // Trigger the scroll event on page load and on scroll window.addEventListener('scroll', animateOnScroll); animateOnScroll(); // Initial check });
